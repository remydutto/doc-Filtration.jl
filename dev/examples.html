Max volume case

Benyahia et al., 2013

In [Benyahia et al., 2013] :

tlim = (0,10); mlim = (0,10)
a = 1; b = 1; e = 1
f₁(m) = b / (e + m)
f₂(m) = a * m
g(m)  = 1 / (e + m)

f₊(m) = 0.5*(f₁(m) + f₂(m))
f₋(m) = 0.5*(f₁(m) - f₂(m))
g₊(m) = g(m)
g₋(m) = 0

model = Model(g₊, g₋, f₊, f₋, :max)
synthesis = Synthesis(model, tlim, mlim)
plot(synthesis, xlabel = "Time (t)", ylabel = "Mass (m)")

Min energy example

Aiouche et al., 2020

In [Aiouche et al., 2020] : 

vlim = (0,50); mlim = (0,50)
ap = 0; bp = 1; cp = 0.16; dp = 3; Qp = 1
ar = 0.5; cr = 0.64; dr = 12; Qr = 2

fp(m) = -ap*m + bp
fr(m) = ar*m
lp(m) = cp*m + dp
lr(m) = cr*m + dr

f₊(m) = 0.5*(fp(m) + fr(m))
f₋(m) = 0.5*(fp(m) - fr(m))
g₊(m) = 0.5*(Qp + Qr)
g₋(m) = 0.5*(Qp - Qr)
e₋(m) = 0.5*(lp(m) + lr(m))
e₊(m) = 0.5*(lp(m) - lr(m))

model = Model(e₊, e₋, f₊, f₋, g₊, g₋, :min; initial_guess = 20.)
synthesis = Synthesis(model, vlim, mlim)
plt = plot(synthesis, xlabel = "Volume (v)", ylabel = "Mass (m)")

Ellouze et al., 2023

In [Ellouze et al., 2023] :

vlim = (0, 0.015); mlim = (0, 0.05)
Jv = 2.71e-6; JBW = 5.66e-6; TMP01 = 7892.4; Rm = 1.79e12
C = 2184e-3; β = 1e-6; α = 2.02e14; η = 4.9e-3
k1f = 2.0769; k2f = 0.0185; k1BW = 9.0551; k2BW = 0.0870

ff(m) = C*Jv - C*Jv*β*m
fBW(m) = -η*m
Ef(m) = k1f * m + k2f
EBW(m) = k1BW * m + k2BW

f₊(m) = 0.5*(ff(m) - fBW(m))
f₋(m) = 0.5*(ff(m) + fBW(m))
g₊(m) = 0.5*(Jv + JBW)
g₋(m) = 0.5*(Jv - JBW)
e₊(m) = 0.5*(Ef(m) - EBW(m))
e₋(m) = 0.5*(Ef(m) + EBW(m))

model = Model(e₊, e₋, f₊, f₋, g₊, g₋, :min; initial_guess = 20.)
synthesis = Synthesis(model, vlim, mlim)
plt = plot(synthesis, SL_nb_points = 10, xlabel = "Volume (v)", ylabel = "Mass (m)")
